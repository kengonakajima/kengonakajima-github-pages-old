 18     var cloneObject = function(obj, deep) {
 19         deep = !!deep;
 20         var ret = {};
 21         var elem;
 22         for (var prop in obj) {
 23             elem = obj[prop];
 24             if (obj.hasOwnProperty(prop)) {
 25                 if (deep) {
 26                     if (elem instanceof Array) {
 27                         ret[prop] = elem.slice();
 28                     } else if (typeof elem === 'object') {
 29                         ret[prop] = cloneObject(elem, deep);
 30                     }
 31                 } else {
 32                     ret[prop] = elem;
 33                 }
 34             }
 35         }
 36         return ret;
 37     };
